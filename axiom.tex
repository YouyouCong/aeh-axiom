\subsection{Axioms}

\begin{defn}
\begin{array}{rcll}
  \app{\absp{x}{e}}{v} & = & 
    \subst{e}{x}{v} &
    \rulename{$\beta_v$} \\
  \abs{x}{\app{v}{x}} & = & 
    v \hspace{5.7cm} \text{$k \not\in \fv{v}$} &
    \rulename{$\eta_v$} \\
  \letin{x}{\ret{v}}{e} & = &
    \subst{e}{x}{v} &
    \rulename{$\beta_{let}$} \\
  \handle{\op{v}}{\hdef} & = &
    \substtwo{\eop}{x}{v}{k}{\abs{x}{\eret}} &
    \rulename{$\beta_h$} \\
  \handle{F[\op{v_1}]}{\opclsone{x}{r}{\app{r}{v_2}}} & = &
    \letin{x}{F[\ret{v_2}]}{\eret} \hspace{2.6cm} \text{$k \notin \fv{v_2}$} &
    \rulename{$\eta_h$} \\ 
  \handle{\ret{v}}{\hdef} & = &
    \subst{\eret}{x}{v} & 
    \rulename{$h_v$} \\   
  \handle{F[e]}{\hdef} & = &
    \handle{e}{\retclsone{x}{\handle{F[\ret{x}]}{\hdef}}} & 
    \rulename{$h_F$}
\end{array}
\end{defn}
