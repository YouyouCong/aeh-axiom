\section{Operations with Continuation}

\subsection{Language}

\begin{defn}
\begin{array}{llcl}
  \text{Computations} & e & :: = &
    \opcont{v}{y}{e}
\end{array}
\end{defn}

\subsection{CPS Translation}

\begin{defn}
\begin{array}{rcl}
  \cpsc{\opcontp{v}{y}{e}} & = & 
    \abstwo{k}{h}{\apphandcont{h}{v}{y}{\cpsc{e}}{k}}
\end{array}
\end{defn}
