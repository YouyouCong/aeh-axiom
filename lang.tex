\begin{array}{llcl}
  \text{Values} & v & ::= &
    x \mid \abs{x}{e} \\ 
  \text{Expressions} & e & ::= &
    \ret{v} \mid \app{v}{v} \mid \letin{x}{e}{e} \\
    & & \mid & \op{v} \mid \handle{e}{h} \\
  \text{Effect Handlers} & h & ::= &
    \retop{x}{e}{x}{r}{e}
\end{array}
\end{defn}
